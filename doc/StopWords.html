def add(word)
        if word.class = String
                self.stop_words << word 
                return true
        else
                return false
        end
end

def get_hash()
        stopword_hash = Hash.new
        self.stop_words.each do |word|
                stopword_hash[word] = true
        end
        return stopword_hash
end

def remove(word)
        if word.class = String and self.stop_words.index(word)
                self.stop_words.delete(word)
                return true
        else
                return false
        end
end
